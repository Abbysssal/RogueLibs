## Making items targetable {#making-targetable}

Just implement the `IItemTargetable` interface in your item's class:

```csharp
// highlight-next-line
public class MyCustomItem : CustomItem, IItemTargetable
{
    // highlight-start
    public bool TargetFilter(PlayfieldObject target) { /*...*/ }
    public bool TargetObject(PlayfieldObject target) { /*...*/ }
    public CustomTooltip TargetCursorText(PlayfieldObject target) { /*...*/ }
    // highlight-end
}
```

`TargetFilter` determines what objects to highlight when using the current item.

`TargetObject` uses the current item on the object. The return value determines whether the usage was successful.

`TargetCursorText` determines the cursor text, that will be visible under the cursor.

:::caution
`TargetFilter` does not control what objects can be passed as parameters to other methods. You need to check if the passed object is compatible inside other methods too:

```csharp
    public bool TargetObject(PlayfieldObject target)
    {
        // highlight-next-line
        if (!TargetFilter(target)) return false;
        // ...
    }
    public CustomTooltip TargetCursorText(PlayfieldObject target) => null;
```
:::

SPYTRON 3000 from Team Fortress 2. Makes you look like the selected person and copies their relationships.

```csharp title="SPYTRON3000.cs"
[ItemCategories(RogueCategories.Social, RogueCategories.Stealth,
    RogueCategories.Technology, RogueCategories.Usable)]
public class SPYTRON3000 : CustomItem, IItemTargetable
{
    public override void SetupDetails()
    {
        Item.itemType = ItemTypes.Tool;
        Item.itemValue = 40;
        Item.initCount = 2;
        Item.rewardCount = 3;
        Item.stackable = true;
        Item.goesInToolbar = true;
    }
    public bool TargetFilter(PlayfieldObject target) => target is Agent a && a != Owner;
    public bool TargetObject(PlayfieldObject targetObj)
    {
        if (!TargetFilter(targetObj)) return false;

        Agent target = (Agent)obj;
        
        string prev = Owner.agentName;
        Owner.agentName = target.agentName;

        Owner.relationships.CopyLooks(target);

        Owner.gc.audioHandler.Play(Owner, "Spawn");
        Owner.gc.spawnerMain.SpawnParticleEffect("Spawn", Owner.tr.position, 0f);

        foreach (Relationship rel in target.relationships.RelList)
        {
            Relationship otherRel = rel.agent.relationships.GetRelationship(target);

            Owner.relationships.SetRel(rel.agent, rel.relType);
            Owner.relationships.SetRelHate(rel.agent, 0);
            Owner.relationships.GetRelationship(rel.agent).secretHate = rel.secretHate;
            Owner.relationships.GetRelationship(rel.agent).mechHate = rel.mechHate;

            rel.agent.relationships.SetRel(Owner, otherRel.relType);
            rel.agent.relationships.SetRelHate(Owner, 0);
            rel.agent.relationships.GetRelationship(Owner).secretHate = otherRel.secretHate;
            rel.agent.relationships.GetRelationship(Owner).mechHate = otherRel.mechHate;
        }

        target.relationships.SetRel(Owner, "Hateful");
        target.relationships.SetRelHate(Owner, 25);
        Owner.agentName = prev;

        Count--;
        Item.invInterface.HideTarget();
    }
    public CustomTooltip TargetCursorText(PlayfieldObject _) => gc.nameDB.GetName("Disguise", "Interface");
}
```

```csharp
RogueLibs.CreateCustomItem<SPYTRON3000>()
    .WithName(new CustomNameInfo("SPYTRON 3000"))
    .WithDescription(new CustomNameInfo("Always wanted to be someone else? Now you can!"))
    .WithSprite(Properties.Resources.SPYTRON3000)
    .WithUnlock(new ItemUnlock { UnlockCost = 10, CharacterCreationCost = 3, LoadoutCost = 2 });

RogueLibs.CreateCustomName("Disguise", "Interface", new CustomNameInfo("Disguise as"));
```

<TabItem value="blankvoodoodoll">

Blank Voodoo Doll. See [Recharging Items: Examples](./recharging-items#examples) for the activated Voodoo Doll.

```csharp title="BlankVoodooDoll.cs"
[ItemCategories(RogueCategories.Usable, RogueCategories.Stealth, RogueCategories.Weird)]
public class BlankVoodooDoll : CustomItem, IItemTargetable
{
    public override void SetupDetails()
    {
        Item.itemType = ItemTypes.Tool;
        Item.itemValue = 100;
        Item.initCount = 3;
        Item.rewardCount = 3;
        Item.stackable = true;
        Item.hasCharges = true;
        Item.goesInToolbar = true;
    }
    public bool TargetFilter(PlayfieldObject target) => target is Agent agent && !agent.dead && !agent.electronic;
    public bool TargetObject(PlayfieldObject target)
    {
        if (!TargetFilter(target)) return false;

        Inventory.DestroyItem(Item);
        VoodooDoll doll = Inventory.AddItem<VoodooDoll>(Count);
        doll.Victim = (Agent)target;
        
        Item.invInterface.HideTarget();
    }
    public CustomTooltip TargetCursorText(PlayfieldObject target)
    {
        if (!(target is Agent agent)) return null;
        if (agent.dead) return gc.nameDB.GetName("VoodooDead", "Interface");
        if (agent.electronic) return gc.nameDB.GetName("VoodooElectronic", "Interface");
        return gc.nameDB.GetName("VoodooBind", "Interface");
    }
}
```

```csharp
RogueLibs.CreateCustomItem<BlankVoodooDoll>()
    .WithName(new CustomNameInfo("Blank Voodoo Doll"))
    .WithDescription(new CustomNameInfo("Bind it to someone first. Has limited uses."))
    .WithSprite(Properties.Resources.BlankVoodooDoll)
    .WithUnlock(new ItemUnlock
    {
        UnlockCost = 10,
        CharacterCreationCost = 5,
        LoadoutCost = 5,
        Prerequisites = { "Necronomicon" },
    });

RogueLibs.CreateCustomName("VoodooBind", "Interface", new CustomNameInfo("Bind"));
RogueLibs.CreateCustomName("VoodooDead", "Interface", new CustomNameInfo("They're already dead"));
RogueLibs.CreateCustomName("VoodooElectronic", "Interface", new CustomNameInfo("This thing doesn't have a soul"));
```
